            if (!state.audioContext) return;
            const osc = state.audioContext.createOscillator();
            const gain = state.audioContext.createGain();
            osc.type = 'square';
            osc.frequency.value = 800;
            gain.gain.value = 0.3;
            osc.connect(gain);
            gain.connect(state.audioContext.destination);
            osc.start();
            osc.stop(state.audioContext.currentTime + 0.1);
        }
       
        function playStartBell() {

            gain.gain.setValueAtTime(0.5, state.audioContext.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, state.audioContext.currentTime + 1.5);
            osc.connect(gain);
            gain.connect(state.audioContext.destination);
            osc.start();
            osc.stop(state.audioContext.currentTime + 1.5);
        }
       
        function playWoodBlock() {
            if (!state.audioContext) return;

                setTimeout(() => {
                    const osc = state.audioContext.createOscillator();
                    const gain = state.audioContext.createGain();
                    const filter = state.audioContext.createBiquadFilter();
                    osc.type = 'square';
                    osc.frequency.value = 800;
                    filter.type = 'lowpass';
                    filter.frequency.value = 1000;
                    gain.gain.value = 0.4;
                    osc.connect(filter);
                    filter.connect(gain);
                    gain.connect(state.audioContext.destination);
                    osc.start();
                    osc.stop(state.audioContext.currentTime + 0.08);
                }, i * 200);
            }
        }
